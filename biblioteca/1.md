```sql
-- Usar 'master' para crear la nueva base de datos
USE master;
GO

-- 1. CREACIÓN DE LA BASE DE DATOS
-- Borra la base de datos si ya existe, para empezar de cero
IF DB_ID('BibliotecaBasica') IS NOT NULL
    DROP DATABASE BibliotecaBasica;
GO

CREATE DATABASE BibliotecaBasica;
GO

-- Indica a SQL Server que queremos usar esta base de datos
USE BibliotecaBasica;
GO

-- 2. CREACIÓN DE TABLAS

-- Tabla de Libros: Almacena nuestro inventario de libros
CREATE TABLE Libros (
    LibroID INT PRIMARY KEY IDENTITY(1,1), -- ID único autoincremental
    Titulo VARCHAR(200) NOT NULL,
    Autor VARCHAR(100) NOT NULL,
    CopiasDisponibles INT NOT NULL -- Cuántas copias tenemos para prestar
);
GO

-- Tabla de Usuarios: Quiénes pueden pedir libros
CREATE TABLE Usuarios (
    UsuarioID INT PRIMARY KEY IDENTITY(1,1),
    NombreCompleto VARCHAR(150) NOT NULL
);
GO

-- Tabla de Préstamos: El historial de quién pidió qué
CREATE TABLE Prestamos (
    PrestamoID INT PRIMARY KEY IDENTITY(1,1),
    LibroID INT FOREIGN KEY REFERENCES Libros(LibroID),     -- FK a la tabla Libros
    UsuarioID INT FOREIGN KEY REFERENCES Usuarios(UsuarioID), -- FK a la tabla Usuarios
    FechaPrestamo DATETIME DEFAULT GETDATE(),               -- Fecha automática del préstamo
    FechaDevolucion DATETIME NULL -- Queda NULO hasta que lo devuelvan
);
GO
```
