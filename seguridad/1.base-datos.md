```sql
-- Crear la base de datos
CREATE DATABASE MiEmpresaDB;
GO

-- Usar la base de datos
USE MiEmpresaDB;
GO

-- Crear un esquema para Recursos Humanos (HR)
CREATE SCHEMA HR;
GO

-- Crear una tabla de empleados dentro del esquema HR
CREATE TABLE HR.Empleados (
    EmpleadoID INT PRIMARY KEY IDENTITY(1,1),
    Nombre VARCHAR(50) NOT NULL,
    Apellido VARCHAR(50) NOT NULL,
    Email VARCHAR(100) UNIQUE,
    FechaContratacion DATE,
    Salario DECIMAL(10, 2)
);
GO

-- Crear un esquema para Finanzas (FN)
CREATE SCHEMA FN;
GO

-- Crear una tabla de facturas dentro del esquema FN
CREATE TABLE FN.Facturas (
    FacturaID INT PRIMARY KEY IDENTITY(1,1),
    EmpleadoID INT,
    Monto DECIMAL(10, 2) NOT NULL,
    FechaFactura DATE,
    FOREIGN KEY (EmpleadoID) REFERENCES HR.Empleados(EmpleadoID)
);
GO
```
